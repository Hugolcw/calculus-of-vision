    """

def calculate_duration(text, base_speed=0.12):
    """
    æ ¹æ®å­—å¹•æ–‡æœ¬è®¡ç®—åŠ¨ç”»æ—¶é•¿
    
    å‚æ•°ï¼š
    - text: å­—å¹•æ–‡æœ¬

    è¿”å›ï¼šå»ºè®®çš„åŠ¨ç”»æ—¶é•¿ï¼ˆç§’ï¼‰
    """
    # åŸºç¡€æ—¶é•¿ï¼šæ¯ä¸ªå­—ç¬¦çš„æ—¶é—´

    # å¤æ‚æ¦‚å¿µï¼ˆæ•°å­¦å…¬å¼ã€ä¸“ä¸šæœ¯è¯­ï¼‰éœ€è¦æ›´å¤šæ—¶é—´

    complexity_penalty = sum(1 for keyword in complex_keywords if keyword in text) * 0.5

    # æœ€å°1ç§’ï¼Œæœ€å¤§ä¸è¶…è¿‡5ç§’ï¼ˆå•å¥å­—å¹•ï¼‰
    return max(1, min(final_time, 5))
